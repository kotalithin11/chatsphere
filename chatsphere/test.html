        function testLogin() {
            const email = document.getElementById('login-email').value;
            const password = document.getElementById('login-password').value;

            try {
                const userProfile = new UserProfile();
                const user = userProfile.login(email, password);
                document.getElementById('login-result').innerHTML = 'Success: Logged in as ' + user.name + '';
            } catch (error) {
                document.getElementById('login-result').innerHTML = 'Error: ' + error.message + '';
            }
        }

        function viewUsers() {
            const users = JSON.parse(localStorage.getItem('chatsphere_users') || '{}');
            document.getElementById('users-data').textContent = JSON.stringify(users, null, 2);
        }

        function clearData() {
            localStorage.removeItem('chatsphere_users');
            localStorage.removeItem('chatsphere_current_user');
            document.getElementById('users-data').textContent = 'Data cleared';
            document.getElementById('register-result').innerHTML = '';
            document.getElementById('login-result').innerHTML = '';
        }
